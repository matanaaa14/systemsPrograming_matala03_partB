    Fraction Fraction::convert(float num)const{
      int Int = 0;
      if(num < 0.001 && num > -0.001)
        return Fraction(0,1);
      if(num >= 1 || num <= -1)
        Int = (int) num;
      int temp = (int)(num * 1000);
      temp = temp % 1000;
      return Fraction(temp + (Int * 1000),1000);
    }
    int Fraction::GCD(int num1, int num2){
      if( num2 == 0)
        return num1;
      return GCD(num2, num1 % num2);
    }
